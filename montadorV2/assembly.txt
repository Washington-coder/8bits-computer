.code
    ; Carrega numero do teclado e guarda em r3
    data r2, 0f
    out addr, r2
    in data, r3
    xor r2, r2
    out addr, r2
    ; Gambiarra para diminuir 0x30 em decimal na tabela ascii
    data r0, 1
    data r1, 0x30
    not r1, r1
    add r1, r0
    add r0, r3
    ; Guarda numero lido do teclado na memoria
    data r0, 0x2f
    st r0, r3
    ; Carregar elementos do vetor no registrador
    data r0, 1
    data r1, 0x2a
    data r2, 0x2e
    clf
    cmp r1, r2
    ja 0x28
        ld r1, r3
        add r0, r1
        ; Guarda a posicao atual no vetor
        data r0, 0x30
        st r0, r1
        data r0, 1
        data r2, 0x27
        data r3, 0x30
        ld r3, r1
        jmp 0x16
    halt
.data

    .word 2
